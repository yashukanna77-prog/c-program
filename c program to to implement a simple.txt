//c program to to implement a simple calculator to perform addtion,subtraction,multipication and divsion based on user's choice.
#include <stdio.h>
int main() {
    float num1, num2, result;
    char op;
    printf("simple calculator using switch case\n");
    printf("________________\n");
    printf("enter an operator (+, -, *, /): ");
    scanf(" %c", &op); 
    printf("enter two numbers: ");
    scanf("%f%f", &num1, &num2);
    
    switch (op) {
        case '+':
            result = num1 + num2;
            printf("Result = %.2f\n", result);
            break;
            
        case '-':
            result = num1 - num2; // Corrected: Used num1 instead of num
            printf("Result = %.2f\n", result);
            break;
            
        case '*':
            result = num1 * num2;
            printf("Result = %.2f\n", result); 
            break;
            
        case '/':
            if (num2 != 0) {
                result = num1 / num2; // Added assignment for consistency, or can print directly
                printf("Result = %.2f\n", result);
            } else {
                printf("Error! Division by zero is not allowed.\n");
            }
            break;
            
        default:
            printf("Invalid operator! Please use +, -, *, or /.\n");
    }
    
    return 0;
}
