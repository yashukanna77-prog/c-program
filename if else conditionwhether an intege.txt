// if else condition:whether an integer is odd or even
#include <stdio.h>

int main() {
    int n;
    printf("enter an integer:");
    scanf("%d",&n);
    if (n%2==0)
    {
        printf("%d is an even integer",n);
        
    }
else {
    printf("%d is odd an integer",n);
    
}
    return 0;
}
